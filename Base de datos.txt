-- 1. Eliminar la base de datos si existe
DROP DATABASE IF EXISTS transporto;

-- 2. Crear nuevamente la base de datos
CREATE DATABASE transporto;
USE transporto;

CREATE TABLE users (
    id INT AUTO_INCREMENT PRIMARY KEY,
    nombre VARCHAR(100),
    email VARCHAR(100) UNIQUE,
    telefono VARCHAR(20),
    fecha_nacimiento DATE,
    contrase√±a VARCHAR(255)
);

CREATE TABLE cooperatives (
    id INT AUTO_INCREMENT PRIMARY KEY,
    nombre VARCHAR(100),
    color VARCHAR(50),
    ciudad VARCHAR(100)
);

CREATE TABLE routes (
    id INT AUTO_INCREMENT PRIMARY KEY,
    cooperativa_id INT,
    descripcion VARCHAR(255),
    tarifa DECIMAL(10,2),
    horario VARCHAR(100),
    FOREIGN KEY (cooperativa_id) REFERENCES cooperatives(id)
);

CREATE TABLE recharges (
    id INT AUTO_INCREMENT PRIMARY KEY,
    user_id INT,
    monto DECIMAL(10,2),
    metodo_pago VARCHAR(100),
    fecha TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (user_id) REFERENCES users(id)
);

CREATE TABLE drivers (
    id INT AUTO_INCREMENT PRIMARY KEY,
    ruta_id INT,
    nombre VARCHAR(100),
    fecha_nacimiento DATE,
    telefono VARCHAR(20),
    estado_civil VARCHAR(50),
    genero VARCHAR(20),
    CONSTRAINT fk_ruta FOREIGN KEY (ruta_id) REFERENCES routes(id)
);

CREATE TABLE saldos (
    id INT AUTO_INCREMENT PRIMARY KEY,
    user_id INT NOT NULL,
    saldo DECIMAL(10, 2) NOT NULL DEFAULT 0,
    FOREIGN KEY (user_id) REFERENCES users(id)
);
